% The standard math packages.
\usepackage{amsmath}
\usepackage{amssymb}
% Used for properly typesetting bold symbols in math mode.
\usepackage{bm}
% Some custom math commands:
% Redefine \vec to use bold font instead of an arrow.
\renewcommand{\vec}[1]{\bm{#1}}
% Provide a command for absolute values.
\newcommand{\abs}[1]{\left\lvert#1\right\rvert}

% Fixes things like bold font.
\usepackage{fontspec}

% Used for links.
\usepackage[pdfusetitle,hidelinks]{hyperref}

% Setup babel for Dutch/English.
\usepackage[dutch, english]{babel}
% And define the date formatting.
\usepackage[us,nodayofweek]{datetime}
% Recommended by babel.
\usepackage{csquotes}

% Use siunitx to write out units and quantities, use special formatting for the units.
\usepackage{siunitx}
\sisetup{separate-uncertainty = true, round-mode = uncertainty, multi-part-units = single, inter-unit-product = \ensuremath { { } \cdot { } }, range-units = single, list-units = single, exponent-product=\cdot}
\robustify{\dots}
\sisetup{input-digits = 0123456789\dots?}

% Setup bibliography (this is a 'major' difference compared to the original LION thesis template).
\usepackage[backend=biber, style=phys, biblabel=brackets]{biblatex}
\addbibresource{sources.bib}

% Let footnotes use numbers instead of symbols.
\renewcommand{\thefootnote}{\arabic{footnote}}

% For double closed integrals among others.
\usepackage{esint}
% Diameter symbol.
\usepackage{wasysym}
% Used for quantum mechanics (braket notation), see https://nl.mirrors.cicku.me/ctan/macros/latex/contrib/braket/braket.pdf.
\usepackage{braket}

% For chemical formulas.
\usepackage[version=4]{mhchem}

% Used for figures.
\usepackage{graphicx}
% Specify how captions should be formatted.
\usepackage[margin=0cm,font={small,sl},labelfont={bf,sl},tableposition=top,singlelinecheck=false,hypcap]{caption}
% Used for subfigures/captions.
\usepackage{subcaption}
% Nicer tables.
\usepackage{booktabs}

% Used for debugging such as printing the value of \textwidth (\printinunitsof{in}\prntlen{\textwidth})
\usepackage{layouts}

% Use whitespace instead of indent for new paragraphs.
\usepackage[parfill]{parskip}
